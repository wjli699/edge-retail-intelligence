[property]
gpu-id=0

# Secondary GIE: runs on each person crop produced by the primary GIE.
# process-mode=2 → secondary (object-level)
# operate-on-gie-id=1 → process objects detected by the primary GIE (PeopleNet)
# operate-on-class-ids=0 → person class only
process-mode=2
# operate-on-gie-id omitted: when SGIE follows the tracker, NvDCF overwrites
# unique_component_id on most objects so gie-id=1 would silently skip them.
# Omitting the field lets DeepStream process all objects; class filtering below
# restricts it to persons only.
operate-on-class-ids=0

# ReID model — ResNet50 (Market-1501 + AI City 156, 256-dim embeddings)
# Input:  3 × 256 × 128  (C × H × W, RGB)
# Output: 256-dim BN-normalised embedding vector (bottleneck layer)
onnx-file=../../models/reid/resnet50_market1501_aicity156.onnx
model-engine-file=../../models/reid/resnet50_market1501_aicity156.onnx_b8_gpu0_fp16.engine

# ImageNet normalization: output = (pixel - mean) / std
# net-scale-factor ≈ 1/avg_std = 1/57.63
# offsets = per-channel ImageNet mean in [0,255] space (R,G,B)
# DeepStream applies: output = net-scale-factor * (pixel - offset)
net-scale-factor=0.017352607709750568
offsets=123.675;116.28;103.53
model-color-format=0
infer-dims=3;256;128

# network-mode: 2=FP16
network-mode=2

# batch-size: max person crops processed per inference call.
# Set to the maximum number of persons expected across all camera frames in one batch.
batch-size=8

# network-type=100 (OTHER/custom): treats output as raw tensors, not class scores.
# output-tensor-meta=1 attaches the raw float tensor to NvDsObjectMeta so the
# pad probe can read the embedding directly.
network-type=100
output-tensor-meta=1

# gie-unique-id must match kReidGieId in pipeline.cpp (currently 2)
gie-unique-id=2

interval=0
